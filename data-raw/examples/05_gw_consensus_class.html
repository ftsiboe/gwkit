gwkit example 05 — gw_consensus_class(): classifying the sign of the risk discount

Instead of averaging a1, we can classify each state’s local risk discount — discount (a1 < 0), ≈ none, or premium (a1 > 0) — under every specification and let gw_consensus_class() take the modal class plus a Queen-contiguity vote that de-speckles isolated flips into a cleaner regional map of where farmland is risk-discounted.

library(data.table); library(sf); library(ggplot2)
source("_setup.R")     # loads gwkit (dev tree if present) + the ERS framework

data(us_state_ag_census)
d <- data.table::copy(us_state_ag_census)
d[, lv            := log(ag_land_value_per_acre)]
d[, base_rate     := fcip_base_rate]
d[, participation := fcip_adoption]
d[, rate_x_part   := base_rate * participation]
d <- d[is.finite(lv) & is.finite(base_rate) & is.finite(participation)]
d_cs <- d[census_year == max(census_year)]

states <- urbnmapr::get_urbn_map("states", sf = TRUE)
states$state_code <- as.integer(states$state_fips)
states <- states[states$state_code %in% d_cs$state_code, ]
specs <- data.table::CJ(dm = gw_distance_metric_names(),
                        kernel = c("gaussian","exponential","bisquare","boxcar","tricube"),
                        sorted = FALSE)

stk <- data.table::rbindlist(lapply(seq_len(nrow(specs)), function(i) {
  r <- tryCatch(estimate_gwr(
        d_cs, unit = "state_code",
        formula = lv ~ base_rate + participation + rate_x_part,
        geometry = states, poly_id = "state_code",
        distance_metric = specs$dm[i], kernel = specs$kernel[i],
        adaptive = TRUE, bw = 12, terms = "base_rate"),
        error = function(e) NULL)
  if (is.null(r)) return(NULL)
  r[term == "base_rate" & estimand == "mean", .(state_code = unit_id, estimate)]
}), fill = TRUE)

thr <- 0.10 * stats::sd(stk$estimate, na.rm = TRUE)     # near-zero band
stk[, discount_class := data.table::fifelse(estimate < -thr, "discount",
                        data.table::fifelse(estimate >  thr, "premium", "~none"))]
optc <- gw_consensus_class(
  stk, unit_col = "state_code", class_col = "discount_class",
  geometry = states, poly_id = "state_code",
  class_levels = c("discount", "~none", "premium"))
optc[, state_code := as.integer(state_code)]
head(optc[, .(state_code, modal_class, modal_agreement, queen_class, queen_order)])

## 4:         13    discount            0.70    discount           1

## 6:         16    discount            0.82    discount           1
pal <- c(discount = "#B2182B", `~none` = "grey85", premium = "#2166AC")

m_mod <- merge(states, optc[, .(state_code, cls = modal_class)], by = "state_code")
m_qn  <- merge(states, optc[, .(state_code, cls = queen_class)],  by = "state_code")

state_class_map(m_mod, "cls",
  title = "Sign of the risk discount: modal class across specs",
  subtitle = "gw_consensus_class() modal_class", legend = "a1 sign", palette = pal)

state_class_map(m_qn, "cls",
  title = "Sign of the risk discount: Queen-vote (de-speckled)",
  subtitle = "gw_consensus_class() queen_class", legend = "a1 sign", palette = pal)

The Queen-vote panel removes lone states whose modal sign disagrees with their entire neighbourhood, yielding contiguous risk-discount regions rather than a speckled map — useful when the story is about where, regionally, farmland carries a risk discount.
